Disini gw coba emulate instruksi aarch64

from unicorn import *
from unicorn.arm64_const import *

# Kode ARM64 dalam bentuk string
ARM64_CODE = (
    b"\xd2\x10\x01\x91"  # mov x8, #0x81
    b"\xe0\x00\x3f\xaa"  # mvn x0, x2
    b"\x21\x00\x00\x91"  # mov x1, #0x9
    b"\xd4\x20\xff\xd3"  # svc #0x1337
)

# Mulai alamat emulasi
ADDRESS = 0x1000

# Inisialisasi mesin emulasi Unicorn untuk ARM64
mu = Uc(UC_ARCH_ARM64, UC_MODE_ARM)

# Alokasikan memori
mu.mem_map(ADDRESS, 0x1000)

# Tulis kode ke memori
mu.mem_write(ADDRESS, ARM64_CODE)

# Atur register
mu.reg_write(UC_ARM64_REG_X8, 0)
mu.reg_write(UC_ARM64_REG_X0, 0)
mu.reg_write(UC_ARM64_REG_X1, 0)
mu.reg_write(UC_ARM64_REG_X2, 0)

# Emulasikan kode mulai dari ADDRESS
try:
    mu.emu_start(ADDRESS, ADDRESS + len(ARM64_CODE))
except UcError as e:
    print(f"ERROR: {e}")

# Baca dan cetak opcode dari kode yang dieksekusi
for i in range(0, len(ARM64_CODE), 4):
    opcode = mu.mem_read(ADDRESS + i, 4)
    print(f"Opcode: {' '.join(f'{b:02x}' for b in opcode)}")
